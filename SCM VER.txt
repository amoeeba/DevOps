::-----------------------------::
::Obtain date and time to create Versioning Package Name.
::This will be different each time an application is versioned.
set datestr=%date:~-4,4%%date:~-10,2%%date:~-7,2%
set hr=%time:~-11,2%
if %hr% lss 10 set hr=0%time:~-10,1%
set mn=%time:~-8,2%
if %mn% lss 10 set mn=0%time:~-7,1%
set sc=%time:~-5,2%
if %sc% lss 10 set sc=0%time:~-4,1%
set ms=%time:~-2,2%
::set timestr=%hr%%mn%%sc%%ms%
set timestr=%hr%%mn%%sc%
@echo %datestr%-%timestr%

::Rename trigger file for respective state:
attrib -r J:\SCMtools\Scripts\Deploy\VER\%ARTNM%.*
ren J:\SCMtools\Scripts\Deploy\VER\%ARTNM%.* %ARTNM%.running

::Deleting this file indicates that there are no issues with the setapp portion of the script.
::Additional error checking is in place in the event of a versioning error.
Del /q J:\SCMtools\Scripts\Deploy\eMail\Artifactversioning.txt
	
::This log documents each step in the process so the results can easily be seen in one log file.
Echo Versioning Log started at:  > J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
Date /t  >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
Time /t  >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
Echo Versioning %LC% for %ARTNM%.%TYPE%: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log 
ECHO ************************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
ECHO Beginning Build of %LC%..... >> J:\SCMtools\Logs\Artifact_versioning.log

ECHO Cleaning up local build content... >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
Del /f /s /q J:\ref_dir\%LC%\INT\*.* >> J:\SCMtools\Logs\VER\%LC%-CleanDir-%datestr%-%timestr%.log
ECHO Local build folder cleanup completed. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
ECHO ************************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	
ECHO Begin Checkout of all content to local folder at J:\ref_dir\%LC%\INT... >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
hco -b %SCMBROKER% -en %LC% -st INT -vp %RP% -cp "J:\ref_dir\%LC%\INT" -pn "Check Out for Browse" -o J:\SCMtools\Logs\VER\%LC%-Artifact-HCO_All-%datestr%-%timestr%.log -s * -usr harvpublic -pw harvpublic -br

	goto CleanSuccess
) else (
	goto CleanFailed
)

:CleanFailed
	ECHO HCO Error Level: %errorlevel% >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO Local Directory Cleanup and HCO process Failed!  Check the log for errors. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO Local Directory Cleanup and HCO process Failed!  Check the log for errors. >> J:\SCMtools\Logs\Artifact_versioning.log
	ECHO ************************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ************************* >> J:\SCMtools\Logs\Artifact_versioning.log
	::This next command tells the script to send SCM an email when a versioning error occurs.
	xcopy /Y J:\SCMtools\Scripts\Deploy\Development\Artifactversioning.txt J:\SCMtools\Scripts\Deploy\eMail
	goto deltrigger ;

:CleanSuccess
	ECHO %LC% INT local checkout successful. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ************************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	if exist J:\ref_dir\%LC%\INT\dist\%ARTNM%.%TYPE% (
	Del /f /s /q J:\ref_dir\%LC%\INT\dist\%ARTNM%.%TYPE% >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO %ARTNM%.%TYPE% Deleted from dist directory. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	) else (
	ECHO Artifact not present in dist folder, no need to delete. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	)
	
	ECHO Create Versioning package and check out %TYPE% for update: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	hcp %ARTNM%-%TYPE%-%datestr%-%timestr% -b %SCMBROKER% -en %LC% -st Versioning -usr kahEQA -pw kahEQA
	hco -b %SCMBROKER% -en %LC% -st Versioning -vp \%RP%\dist -p %ARTNM%-%TYPE%-%datestr%-%timestr% -cp J:\ref_dir\%LC%\INT\dist -pn "Check Out for Update" -s %ARTNM%.%TYPE% -o J:\SCMtools\Logs\INT\%LC%_%TYPE%_HCO-Updt.log -usr kahEQA -pw kahEQA -up
	if not exist J:\ref_dir\%LC%\INT\dist\%ARTNM%.%TYPE% goto Fail-1
	del /f /s /q J:\ref_dir\%LC%\INT\dist\%ARTNM%.%TYPE% >>J:\SCMtools\Logs\INT\%LC%_BUILD.log
	ECHO Package creation and %ARTNM%.%TYPE% checkout for update completed successfully. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ************************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	
	ECHO Begin build of new %ARTNM%.%TYPE% for %LC%: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	CD J:\ref_dir\%LC%\INT\%ARTNM%Build
	CALL build 1>"J:\SCMtools\Logs\VER\%LC%_BuildOutput-%datestr%-%timestr%.log" 2>"J:\SCMtools\Logs\VER\%LC%_BuildError-%datestr%-%timestr%.log"
	TYPE "J:\SCMtools\Logs\VER\%LC%_BuildOutput-%datestr%-%timestr%.log" >>J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	TYPE "J:\SCMtools\Logs\VER\%LC%_BuildError-%datestr%-%timestr%.log" >>J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ********************* >>J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	if not exist J:\ref_dir\%LC%\INT\dist\%ARTNM%.%TYPE% goto Fail-2
	
	ECHO Check new %TYPE% back into Harvest: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	hci -b %SCMBROKER% -en %LC% -st Versioning -p %ARTNM%-%TYPE%-%datestr%-%timestr% -vp \%RP%\dist -pn "Check In New and Existing Items" -cp J:\ref_dir\%LC%\INT\dist -s %ARTNM%.%TYPE% -o J:\SCMtools\Logs\INT\%LC%_%TYPE%_HCI.log -usr kahEQA -pw kahEQA -ur
	::hci %LC%_hco_CleanDIR.log -b %SCMBROKER% -en %LC% -st Versioning -p %ARTNM%-%TYPE%-%datestr%-%timestr% -vp \%LC%\dist\hlogs -pn "Check In New and Existing Items" -cp J:\SCMtools\Logs\INT -s %LC%_hco_CleanDIR.log -if ne -o J:\SCMtools\Logs\INT\%LC%_CleanDIR_HCI.log -usr kahEQA -pw kahEQA -ur >>J:\SCMtools\Logs\INT\%LC%_BUILD.log
	ECHO %TYPE% Check-in Process Complete: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ******************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO %TYPE% Check-in Process Complete. >> J:\SCMtools\Logs\Artifact_versioning.log
	ECHO Versioning Process Finished at: >> J:\SCMtools\Logs\Artifact_versioning.log
	time /t >> J:\SCMtools\Logs\Artifact_versioning.log
	ECHO *----------------------------------------* >> J:\SCMtools\Logs\Artifact_versioning.log
	
	ECHO Promote Versioning package to INT... >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	hpp %ARTNM%-%TYPE%-%datestr%-%timestr% -b %SCMBROKER% -en %LC% -st Versioning -usr kahEQA -pw kahEQA -pn "Promote to INT"
	ECHO Package promotion successful.  A deploy has been triggered and will take place shortly. >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO %LC% versioning process finished at: >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	time /t >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	ECHO ******************** >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	
	::This next command sends an email to the application dev team stating that versioning
	::has completed and that a deploy has been triggered.
	J:\SCMtools\email\tcmail -s "Versioning of %ARTNM%.%TYPE% in INT for %LC% Successful.  See message body for build output." %MAIL% < J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
	::This next command tells the script to send SCM an email when versioning completes.
	xcopy /Y J:\SCMtools\Scripts\Deploy\Development\VersioningEnd.txt J:\SCMtools\Scripts\Deploy\eMail
	goto deltrigger ;

:Fail-1
::--------------------------------------
Echo "ERROR - %ARTNM%.%TYPE% not found after HCO for Update Process.  Check %LC% Lifecycle for problems." >>J:\SCMtools\Logs\Artifact_versioning.log
Echo "ERROR - %ARTNM%.%TYPE% not found after HCO for Update Process.  Check %LC% Lifecycle for problems." J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log

time /t >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
ECHO ******************* >> J:\SCMtools\Logs\VER\%LC%-Versioning-%datestr%-%timestr%.log
